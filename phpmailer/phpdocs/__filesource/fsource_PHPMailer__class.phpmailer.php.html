    433. 

  433.         //$result = false;
    434. 

  434.         //break;
    435. 

  435.     }

  445.   function SendmailSend($header, $body) {
    446. 

  446.     if ($this->Sender != '') {
    447. 

  447.       $sendmail = sprintf("%s -oi -f %s -t", escapeshellcmd($this->Sendmail), escapeshellarg($this->Sender));
    448. 

  448.     } else {
    449. 

  449.       $sendmail = sprintf("%s -oi -t", escapeshellcmd($this->Sendmail));
    450. 

  450.     }
    451. 

  451.  
    452. 

  452.     if(!@$mail = popen($sendmail, 'w')) {
    453. 

  453.       $this->SetError($this->Lang('execute') . $this->Sendmail);
    454. 

  454.       return false;
    455. 

  455.     }
    456. 

  456.  
    457. 

  457.     fputs($mail, $header);
    458. 

  458.     fputs($mail, $body);
    459. 

  459.  
    460. 

  460.     $result = pclose($mail);
    461. 

  461.     if (version_compare(phpversion(), '4.2.3') == -1) {
    462. 

  462.       $result = $result >> 8 & 0xFF;
    463. 

  463.     }
    464. 

  464.     if($result != 0) {
    465. 

  465.       $this->SetError($this->Lang('execute') . $this->Sendmail);
    466. 

  466.       return false;
    467. 

  467.     }
    468. 

  468.     return true;
    469. 

  469.   }

    811.

    813. 

  813.    * @return int

    815. 

  815.   function UTF8CharBoundary($encodedText, $maxLength) {
    816.

    817. 

  817.     $lookBack = 3;

  818.     while (!$foundSplitPos) {

  826.         if ($dec < 128) { // Single byte character.
    827. 

  827.           // If the encoded char was found at pos 0, it will fit
    828. 

  828.           // otherwise reduce maxLength to start of the encoded char

  829.           $maxLength = ($encodedCharPos == 0) ? $maxLength :
    830. 

  830.           $maxLength - ($lookBack - $encodedCharPos);
    831. 

  831.           $foundSplitPos = true;
    832. 

  832.         } elseif ($dec >= 192) { // First byte of a multi byte character
    833. 

  833.           // Reduce maxLength to split at start of character
    834. 

  834.           $maxLength = $maxLength - ($lookBack - $encodedCharPos);
    835. 

  835.           $foundSplitPos = true;
    836. 

  836.         } elseif ($dec < 192) { // Middle byte of a multi byte character, look further back
    837. 

  837.           $lookBack += 3;
    838. 

  838.         }
    839. 

  839.       } else {
    840. 

  840.         // No encoded character found
    841. 

  841.         $foundSplitPos = true;
    842. 

  842.       }
    843. 

  843.     }
    844. 

  844.     return $maxLength;
