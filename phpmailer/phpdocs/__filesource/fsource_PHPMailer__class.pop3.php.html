  134.   function Authorise ($host, $port = false, $tval = false, $username, $password, $debug_level = 0)
    135. 

  135.   {
    136. 

  136.     $this->host = $host;
    137. 

  137.  
    138. 

  138.     //  If no port value is passed, retrieve it
    139. 

  139.     if ($port == false)
    140. 

  140.     {
    141. 

  141.       $this->port = $this->POP3_PORT;
    142. 

  142.     }
    143. 

  143.     else
    144. 

  144.     {
    145. 

  145.       $this->port = $port;
    146. 

  146.     }
    147. 

  147.  
    148. 

  148.     //  If no port value is passed, retrieve it
    149. 

  149.     if ($tval == false)
    150. 

  150.     {
    151. 

  151.       $this->tval = $this->POP3_TIMEOUT;
    152. 

  152.     }
    153. 

  153.     else
    154. 

  154.     {
    155. 

  155.       $this->tval = $tval;
    156. 

  156.     }
    157. 

  157.  
    158. 

  158.     $this->do_debug = $debug_level;
    159. 

  159.     $this->username = $username;
    160. 

  160.     $this->password = $password;
    161. 

  161.  
    162. 

  162.     //  Refresh the error log
    163. 

  163.       $this->error = null;
    164. 

  164.  
    165. 

  165.       //  Connect
    166. 

  166.     $result = $this->Connect($this->host, $this->port, $this->tval);
    167. 

  167.  
    168. 

  168.     if ($result)
    169. 

  169.     {
    170. 

  170.       $login_result = $this->Login($this->username, $this->password);
    171. 

  171.  
    172. 

  172.       if ($login_result)
    173. 

  173.       {
    174. 

  174.         $this->Disconnect();
    175. 

  175.  
    176. 

  176.         return true;
    177. 

  177.       }
    178. 

  178.  
    179. 

  179.     }
    180. 

  180.  
    181. 

  181.     //  We need to disconnect regardless if the login succeeded
    182. 

  182.     $this->Disconnect();
    183. 

  183.  
    184. 

  184.     return false;
    185. 

  185.   }

  187.   /**
    188. 

  188.    * Connect to the POP3 server
    189. 

  189.    *
    190. 

  190.    * @param string $host 
    191. 

  191.    * @param integer $port 
    192. 

  192.    * @param integer $tval 
    193. 

  193.    * @return boolean 
    194. 

  194.    */
    195. 

  195.   function Connect ($host, $port = false, $tval = 30)
    196. 

  196.     {
    197. 

  197.     //  Are we already connected?
    198. 

  198.     if ($this->connected)
    199. 

  199.     {
    200. 

  200.       return true;
    201. 

  201.     }
    202. 

  202.  
    203. 

  203.     /*
    204. 

  204.       On Windows this will raise a PHP Warning error if the hostname doesn't exist.
    205. 

  205.       Rather than supress it with @fsockopen, let's capture it cleanly instead
    206. 

  206.     */
    207. 

  207.  
    208. 

  208.     set_error_handler(array(&$this, 'catchWarning'));
    209. 

  209.  
    210. 

  210.     //  Connect to the POP3 server
    211. 

  211.     $this->pop_conn = fsockopen($host,    //  POP3 Host
    212. 

  212.                   $port,    //  Port #
    213. 

  213.                   $errno,   //  Error Number
    214. 

  214.                   $errstr,  //  Error Message
    215. 

  215.                   $tval);   //  Timeout (seconds)
    216. 

  216.  
    217. 

  217.     //  Restore the error handler
    218. 

  218.     restore_error_handler();
    219. 

  219.  
    220. 

  220.     //  Does the Error Log now contain anything?
    221. 

  221.     if ($this->error && $this->do_debug >= 1)
    222. 

  222.     {
    223. 

  223.         $this->displayErrors();
    224. 

  224.     }
    225. 

  225.  
    226. 

  226.     //  Did we connect?
    227. 

  227.       if ($this->pop_conn == false)
    228. 

  228.       {
    229. 

  229.         //  It would appear not...
    230. 

  230.         $this->error = array(
    231. 

  231.           'error' => "Failed to connect to server $host on port $port",
    232. 

  232.           'errno' => $errno,
    233. 

  233.           'errstr' => $errstr
    234. 

  234.         );
    235. 

  235.  
    236. 

  236.         if ($this->do_debug >= 1)
    237. 

  237.         {
    238. 

  238.           $this->displayErrors();
    239. 

  239.         }
    240. 

  240.  
    241. 

  241.         return false;
    242. 

  242.       }
    243. 

  243.  
    244. 

  244.       //  Increase the stream time-out
    245. 

  245.  
    246. 

  246.       //  Check for PHP 4.3.0 or later
    247. 

  247.       if (version_compare(phpversion(), '4.3.0', 'ge'))
    248. 

  248.       {
    249. 

  249.         stream_set_timeout($this->pop_conn, $tval, 0);
    250. 

  250.       }
    251. 

  251.       else
    252. 

  252.       {
    253. 

  253.         //  Does not work on Windows
    254. 

  254.         if (substr(PHP_OS, 0, 3) !== 'WIN')
    255. 

  255.         {
    256. 

  256.           socket_set_timeout($this->pop_conn, $tval, 0);
    257. 

  257.         }
    258. 

  258.       }
    259. 

  259.  
    260. 

  260.     //  Get the POP3 server response
    261. 

  261.       $pop3_response = $this->getResponse();
    262. 

  262.  
    263. 

  263.       //  Check for the +OK
    264. 

  264.       if ($this->checkResponse($pop3_response))
    265. 

  265.       {
    266. 

  266.       //  The connection is established and the POP3 server is talking
    267. 

  267.       $this->connected = true;
    268. 

  268.         return true;
    269. 

  269.       }
    270. 

  270.  
    271. 

  271.     }

  273.     /**
    274. 

  274.      * Login to the POP3 server (does not support APOP yet)
    275. 

  275.      *
    276. 

  276.      * @param string $username 
    277. 

  277.      * @param string $password 
    278. 

  278.      * @return boolean 
    279. 

  279.      */
    280. 

  280.     function Login ($username = '', $password = '')
    281. 

  281.     {
    282. 

  282.       if ($this->connected == false)
    283. 

  283.       {
    284. 

  284.         $this->error = 'Not connected to POP3 server';
    285. 

  285.  
    286. 

  286.         if ($this->do_debug >= 1)
    287. 

  287.         {
    288. 

  288.           $this->displayErrors();
    289. 

  289.         }
    290. 

  290.       }
    291. 

  291.  
    292. 

  292.       if (empty($username))
    293. 

  293.       {
    294. 

  294.         $username = $this->username;
    295. 

  295.       }
    296. 

  296.  
    297. 

  297.       if (empty($password))
    298. 

  298.       {
    299. 

  299.         $password = $this->password;
    300. 

  300.       }
    301. 

  301.  
    302. 

  302.     $pop_username = "USER $username" . $this->CRLF;
    303. 

  303.     $pop_password = "PASS $password" . $this->CRLF;
    304. 

  304.  
    305. 

  305.       //  Send the Username
    306. 

  306.       $this->sendString($pop_username);
    307. 

  307.       $pop3_response = $this->getResponse();
    308. 

  308.  
    309. 

  309.       if ($this->checkResponse($pop3_response))
    310. 

  310.       {
    311. 

  311.         //  Send the Password
    312. 

  312.         $this->sendString($pop_password);
    313. 

  313.         $pop3_response = $this->getResponse();
    314. 

  314.  
    315. 

  315.         if ($this->checkResponse($pop3_response))
    316. 

  316.         {
    317. 

  317.           return true;
    318. 

  318.         }
    319. 

  319.         else
    320. 

  320.         {
    321. 

  321.           return false;
    322. 

  322.         }
    323. 

  323.       }
    324. 

  324.       else
    325. 

  325.       {
    326. 

  326.         return false;
    327. 

  327.       }
    328. 

  328.     }

  390.         if ($this->do_debug >= 1)

  391.         {
    392. 

  392.           $this->displayErrors();
    393. 

  393.         }
    394. 

  394.  
    395. 

  395.         return false;
    396. 

  396.       }
    397. 

  397.       else
    398. 

  398.       {
    399. 

  399.         return true;
    400. 

  400.       }
    401. 

  401.  
    402. 

  402.     }
